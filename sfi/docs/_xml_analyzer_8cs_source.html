36 public async override ValueTask<AnalysisResult> Analyze(XmlReader reader, AnalysisContext context, IEntityAnalyzers analyzers)
37 {
38 var node = GetNode(context);
39
40 XDocumentType? docType = null;
41 do
42 {
43 switch(reader.NodeType)
44 {
45 case XmlNodeType.XmlDeclaration:
46 var version = reader.GetAttribute("version");
47 if(version != null)
48 {
49 node.Set(Properties.XmlVersion, version);
50 }
51 var encoding = reader.GetAttribute("encoding");
52 if(encoding != null)
53 {
54 node.Set(Properties.XmlEncoding, encoding);
55 }
56 var standalone = reader.GetAttribute("standalone");
57 if(standalone != null)
58 {
59 node.Set(Properties.XmlStandalone, standalone);
60 }
61 break;
62 case XmlNodeType.DocumentType:
63 var name = reader.Name;
64 var pubid = reader.GetAttribute("PUBLIC");
65 var sysid = reader.GetAttribute("SYSTEM");
66 if(!String.IsNullOrEmpty(pubid))
67 {
68 var dtd = node["#dtd()"];
69 node.Set(Properties.DtDecl, dtd);
70 dtd.SetClass(Classes.DoctypeDecl);
71 if(name != null)
72 {
73 dtd.Set(Properties.DoctypeName, name);
74 }
75 dtd.Set(Properties.PublicId, pubid);
76 // A URI is created from the PUBLIC identifier
77 var dtdParent = context.NodeFactory.Create(UriTools.PublicIdFormatter, pubid);
78 if(dtdParent != null)
79 {
80 dtd.Set(Properties.SeeAlso, dtdParent);
81 }
82 if(sysid != null)
83 {
84 dtd.Set(Properties.SystemId, sysid, Datatypes.AnyUri);
85 var definedNode = context.NodeFactory.Create(UriFormatter.Instance, sysid);
86 if(dtdParent != null && definedNode != null)
87 {
88 // The SYSTEM identifier can be used to define the DTD
89 dtdParent.Set(Properties.IsDefinedBy, definedNode);
90 }
91 }
92 }
93 docType = new XDocumentType(name, pubid, sysid, reader.Value);
94 break;
95 case XmlNodeType.Element:
96 // Describe the root element using the XIS vocabulary
97 var elem = node["#element(/1)"];
98 node.Set(Properties.DocumentElement, elem);
99 elem.SetClass(Classes.Element);
100 elem.Set(Properties.LocalName, reader.LocalName);
101 if(!String.IsNullOrEmpty(reader.Prefix))
102 {
103 elem.Set(Properties.XmlPrefix, reader.Prefix);
104 }
105 var xmlName = reader.Name;
106 elem.Set(Properties.XmlName, xmlName);
107 if(!String.IsNullOrEmpty(reader.NamespaceURI))
108 {
109 elem.Set(Properties.NamespaceName, reader.NamespaceURI, Datatypes.AnyUri);
110 try{
111 elem.Set(Properties.SeeAlso, UriFormatter.Instance, reader.NamespaceURI);
112 }catch(UriFormatException)
113 {
114 try{
115 elem.Set(Properties.SeeAlso, UriTools.PublicIdFormatter, XmlConvert.VerifyPublicId(reader.NamespaceURI));
116 }catch(XmlException)
117 {
118
119 }
120 }
121 }
122
123 context = context.WithNode(null);
124
125 // Capture the state of the XmlReader
126 var rootState = new XmlReaderState(reader);
127
128 // Filter out formats not matching the DTD or root element
129 var formats = XmlFormats.Where(fmt => fmt.CheckDocument(docType, rootState)).ToList();

131 async Task<bool> MatchFormat(IXmlDocumentFormat format, XmlReader localReader)
132 {
133 var result = await format.Match(localReader, docType, context.MatchContext, this, (format, context, analyzers));
134 if(result.Node != null)
135 {
136 node.Set(Properties.HasFormat, result.Node);
137 return true;
138 }
139 return false;
140 }
141
142 bool any = false;
143 if(formats.Count <= 1)
144 {
145 foreach(var format in formats)
146 {
147 // For a single XML format, just give the reader to it
148 if(await MatchFormat(format, reader))
149 {
150 any = true;
151 }
152 }
153 }else{
154 // For multiple formats, each shall get its own channel of replicated XML states
155 var tasks = new Task<bool>[formats.Count];
156 var writers = new ChannelWriter<XmlReaderState>[formats.Count];
157 for(int i = 0; i < formats.Count; i++)
158 {
159 var channelReader = ChannelXmlReader.Create(reader, out writers[i]);
160 var format = formats[i];
161 tasks[i] = Task.Run(() => MatchFormat(format, channelReader));
162 }
163
164 foreach(var state in new[] { rootState }.Concat(XmlReaderState.ReadFrom(reader)))
165 {
166 foreach(var writer in writers)
167 {
168 await writer.WriteAsync(state);
169 }
170 }
171 foreach(var writer in writers)
172 {
173 writer.TryComplete();
174 }
175
176 await Task.WhenAll(tasks);
177
178 any = tasks.Any(t => t.Result);
179 }
180 if(!any)
181 {
182 await MatchFormat(ImprovisedXmlFormat.Instance, rootState);
183 }
184
185 return new AnalysisResult(node, xmlName);
186 }
187 }while(await ReadSafe(reader));
188
189 throw new InvalidOperationException();
190 }
