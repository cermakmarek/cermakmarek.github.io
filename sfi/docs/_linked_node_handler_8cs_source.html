159 private INode CreateNode(Uri uri, IRdfHandler handler)
160 {
161 if(uri == null)
162 {
163 return null!;
164 }
165 if(uri.Scheme == BlankUriScheme)
166 {
167 return CreateBlankNode(uri, handler);
168 }
169 if(uri.OriginalString.Length > MaxUriLength)
170 {
171 // Shorten the URI for displaying
172 var shortUriLabel = UriTools.ShortenUri(uri, UriPartShortened, "\u00A0(URI\u00A0too\u00A0long)");
173
174 var newUri = UriTools.UriToUuidUri(uri);
175 var subject = handler.CreateUriNode(newUri);
176 realUriCache.Add(subject, uri);
177 var node = new UriNode(subject, handler, GetGraphCache(handler));
178 node.Set(Properties.AtPrefLabel, shortUriLabel.ToString(), Datatypes.AnyUri);
179 var linked = node.In(Graphs.ShortenedLinks);
180 if(linked != null)
181 {
182 // Link the individuals in the ShortenedLinks graph
183 linked.Set(Properties.SameAs, UriFormatter.Instance, uri);
184 }
185 return subject;
186 }

314 protected override void HandleTriple(INode? subj, INode? pred, INode? obj)
315 {
316 if(subj == null || pred == null || obj == null)
317 {
318 return;
319 }
320 var date = DateTime.UtcNow;
321 var oldPred = pred;
322 while(pred is IUriNode uriPred && !handler.IsSafePredicate(uriPred.Uri))
323 {
324 // Create v5 UUID URI to encode the unsafe predicate URI
325 pred = Graph.CreateUriNode(UriTools.UriToUuidUri(uriPred.Uri));
326 }
327 lock(Graph)
328 {
329 Graph.HandleTriple(new Triple(subj, pred, obj));
330 }
331 if(pred != oldPred)
332 {
333 var predNode = CreateNew(pred);
334 var oldPredNode = CreateNew(oldPred);
335 predNode.Set(Properties.SameAs, oldPredNode);
336 predNode.Set(Properties.EquivalentProperty, oldPredNode);
337 }
338 var meta = In(Graphs.Metadata);
339 if(meta != null && !Equals(meta))
340 {
341 // Add at:visited to the metadata graph (if this is not the metadata graph)
342 var dateString = XmlConvert.ToString(date, "yyyy-MM-dd\\THH:mm:ssK");
343 meta.Set(Properties.Visited, dateString, Datatypes.DateTime);
344 }

661 private INode? ReplaceNode(INode node)
662 {
663 if(node is IUriNode uriNode)
664 {
665 var nodeUri = uriNode.Uri;
666 if(subjectUris.Any(uri => comparer.Equals(nodeUri, uri)))
667 {
668 return describingNode.Subject;
669 }
670 if(nodeUri.IsAbsoluteUri && nodeUri.Scheme == BaseUri.Scheme)
671 {
672 var idUri = BaseUri.MakeRelativeUri(nodeUri);
673 if(!idUri.IsAbsoluteUri)
674 {
675 var id = idUri.OriginalString;
676 if(id == "")
677 {
678 return describingNode.Subject;
679 }
680 return (describingNode[id] as UriNode)?.Subject;
681 }
682 }
683 }
684 return node;
