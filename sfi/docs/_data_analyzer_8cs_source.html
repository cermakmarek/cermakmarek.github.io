110 public int GetMaxDataLengthToStore(long dataSize)
111 {
112 var min = MinDataLengthToStore;
113 var hashedSize = 0;
114 var formatter = ContentUriFormatter;
115 if(formatter != null)
116 {
117 // The first supported algorithm used by this analyzer will be chosen
118 foreach(var algorithm in formatter.SuitableAlgorithms.OfType<IDataHashAlgorithm>())
119 {
120 if(HashAlgorithms.Contains(algorithm))
121 {
122 var size = algorithm.GetHashSize(dataSize);
123 if(formatter.EstimateUriSize(algorithm, size) is int uriSize)
124 {
125 hashedSize = uriSize;
126 break;
127 }
128 }
129 }
130 }
131
132 hashedSize += HashAlgorithms.Sum(h => {
133 var hashSize = h.GetHashSize(dataSize);
134 // Each hash contributes several triples, its URI, and its value in base64 (a 4/3 increase)
135 return HashAlgorithm.TriplesPerHash * TripleSizeEstimate + h.EstimateUriSize(hashSize) + (hashSize + 2) / 3 * 4;
136 });
137
138 return Math.Max(min, (hashedSize - "data:;base64,".Length - TripleSizeEstimate) * 3 / 8);
139 }

192 public async Task<TResult> Match<TResult>(Func<DataMatch, ValueTask<TResult>> receiver)
193 {
194 var lazyMatch = new Lazy<Task<DataMatch>>(() => DataMatch.Create(this), false);
195
196 long actualLength = 0;
197
198 DataMatch match;
199
200 var tmpPath = default(FileTools.TemporaryFile);
201 try{
202 Stream? outputStream = null;
203 try{
204 using var stream = streamFactory.Open();
205
206 if(streamFactory.Access != StreamFactoryAccess.Parallel || !stream.CanSeek)
207 {
208 if(streamFactory.Length >= analyzer.FileSizeToWriteToDisk)
209 {
210 tmpPath = FileTools.GetTemporaryFile("b");
211 outputStream = new FileStream(tmpPath, FileMode.CreateNew, FileAccess.Write, FileShare.None);
212 }else{
213 outputStream = new MemoryStream();
214 }
215
216 foreach(var hash in analyzer.HashAlgorithms)
217 {
218 var queue = ChannelArrayStream.Create(out var writer, 1);
219 hashes[hash] = (writer, Task.Run(async () => await hash.ComputeHash(queue, streamFactory)));
220 }
221 }else{
222 seekableFactory = streamFactory;
223
224 foreach(var hash in analyzer.HashAlgorithms)
225 {
226 hashes[hash] = (null, Task.Run(async () => {
227 using var hashStream = streamFactory.Open();
228 return await hash.ComputeHash(hashStream, streamFactory);
229 }));
230 }
231 }
232
233 bool? couldBeUnicode = null;
234
235 using var bufferLease = ArrayPool<byte>.Shared.Rent(16384, out var buffer);
236
237 var activeHashes = (IEnumerable<(ChannelWriter<ArraySegment<byte>> writer, Task<byte[]> data)>)hashes.Values.Where(v => v.writer != null);
238
239 int read;
240 while((read = await stream.ReadAsync(buffer, 0, buffer.Length)) != 0)
241 {
242 var segment = buffer.Slice(0, read);
243 async Task WriteToHasher(ChannelWriter<ArraySegment<byte>> writer)
244 {
245 await writer.WriteAsync(segment).ConfigureAwait(false);
246 await writer.WriteAsync(default).ConfigureAwait(false);
247 await writer.WaitToWriteAsync().ConfigureAwait(false);
248 }
249 var writing = activeHashes.Select(hash => Task.WhenAny(WriteToHasher(hash.writer), hash.data)).ToArray();
250
251 actualLength += read;
252
253 if(couldBeUnicode == null)
254 {
255 couldBeUnicode = DataTools.FindBom(buffer.AsSpan(0, read)) > 0;
256 }
257
258 if(!isBinary)
259 {
260 var data = buffer.Slice(0, read);
261 if(couldBeUnicode == false && DataTools.IsBinary(data))
262 {
263 isBinary = true;
264 }else{
265 encodingDetector?.Write(data);
266 }
267 }
268
269 outputStream?.Write(buffer, 0, read);
270
271 if(signatureBuffer.Length < signatureBuffer.Capacity)
272 {
273 signatureBuffer.Write(buffer, 0, Math.Min(signatureBuffer.Capacity - (int)signatureBuffer.Length, read));
274 }else if(outputStream == null)
275 {
276 _ = lazyMatch.Value;
277 }
278
279 await Task.WhenAll(writing);
280 }
281
282 foreach(var hash in activeHashes)
283 {
284 hash.writer.Complete();
285 }
286
287 encodingDetector?.End();
288 }finally{
289 outputStream?.Dispose();
290 }
291
292 if(tmpPath != null)
293 {
294 seekableFactory = new FileInfoWrapper(new FileInfo(tmpPath), streamFactory);
295 }else if(outputStream is MemoryStream memoryStream)
296 {
297 var buffer = memoryStream.GetData();
298 seekableFactory = new MemoryStreamFactory(buffer, streamFactory);
299 }
300
301 match = await lazyMatch.Value;
302 match.ActualLength = actualLength;
303
304 return await receiver(match);
305 }finally{
306 tmpPath.Dispose();
307 }
