164 public Complex[] CreatePolarMatrix(ChannelSpectrum<Complex> spectrum)
165 {
166 var matrix = new Complex[Width * Height];
167 for(int y = 0; y < Height; y++)
168 {
169 for(int x = 0; x < Width; x++)
170 {
171 var pt = new Complex((double)x / Width * 2 - 1, (double)y / Height * 2 - 1);
172
173 var mag = pt.Magnitude;
174 var phase = (pt.Phase / Math.PI + 1) / 2 % 1;
175
176 var sx = (int)Math.Round(phase * spectrum.Count);
177
178 Complex c;
179 if(sx < 0 || sx >= spectrum.Count)
180 {
181 c = Complex.Zero;
182 }else{
183 var row = spectrum[sx];
184 var sy = (int)Math.Round(mag * row.Count);
185 if(sy <= 0 || sy >= row.Count)
186 {
187 c = Complex.Zero;
188 }else{
189 var coef = 1 / mag / Math.PI;
190 c = row.At(sy) / 256;
191 if(c.Magnitude > 1) c /= c.Magnitude;
192 c = coef * c;
193 }
194 }
195 matrix[y * Width + x] = c;
196 }
