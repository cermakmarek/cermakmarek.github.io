20 public async override ValueTask<AnalysisResult> Analyze(ExifDirectoryBase directory, AnalysisContext context, IEntityAnalyzers analyzers)
21 {
22 var node = GetNode(context);
23 node.SetClass(Classes.IFD);
24 foreach(var tag in directory.Tags)
25 {
26 if(!exifFields.TryGetValue(tag.Type, out var id))
27 {
28 continue;
29 }
30 string value;
31 DatatypeUri datatype;
32 var property = GetProperty(id);
33 switch(directory.GetObject(tag.Type))
34 {
35 case Rational r:
36 r = r.GetSimplifiedInstance();
37 if(r.Denominator == 1)
38 {
39 node.Set(property, r.Numerator);
40 continue;
41 }
42 value = r.GetSimplifiedInstance().ToString();
43 datatype = Datatypes.Rational;
44 break;
45 case StringValue s:
46 value = s.ToString();
47 datatype = Datatypes.String;
48 break;
49 case byte[] bs:
50 if(bs.Length <= 8)
51 {
52 var sb = new StringBuilder();
53 foreach(byte b in bs)
54 {
55 sb.Append(b.ToString("X2"));
56 }
57 value = sb.ToString();
58 datatype = Datatypes.HexBinary;
59 }else{
60 value = Convert.ToBase64String(bs);
61 datatype = Datatypes.Base64Binary;
62 }
63 break;
64 case ValueType v:
65 node.TrySet(property, v);
66 continue;
67 case object o:
68 var node2 = (await analyzers.TryAnalyze(o, context)).Node;
69 if(node2 != null)
70 {
71 node.Set(property, node2);
72 }
73 continue;
74 default:
75 continue;
76 }
77 if(String.IsNullOrWhiteSpace(value)) continue;
78 if(datatype == Datatypes.String && DateTime.TryParseExact(value, "yyyy:MM:dd HH:mm:ss", CultureInfo.InvariantCulture, DateTimeStyles.None, out var dateTime))
79 {
80 node.Set(property, dateTime);
81 }else{
82 node.Set(property, value, datatype);
83 }
84 }
85 return new AnalysisResult(node);
86 }
