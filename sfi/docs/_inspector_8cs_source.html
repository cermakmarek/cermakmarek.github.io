201 public async ValueTask Inspect<T>(IEnumerable<T> entities, Func<string, Stream> outputFactory, InspectorOptions options) where T : class
202 {
203 if(options == null) throw new ArgumentNullException(nameof(options));
204
205 // Each custom graph can have a specific handler
206 var graphHandlers = new Dictionary<Uri, IRdfHandler?>(new UriComparer());
207
208 if(options.HideMetadata)
209 {
210 graphHandlers[new Uri(Graphs.Metadata.Value)] = null;
211 }
212
213 graphHandlers[new Uri(Graphs.ShortenedLinks.Value)] = null;
214
215 // Loads SPARQL queries to evaluate on the RDF
216 var queries = await GetQueries(options);
217
218 bool sparql = options.OutputIsSparqlResults;
219
220 var formatStr = options.Format ?? (sparql ? MimeTypesHelper.DefaultSparqlXmlExtension : MimeTypesHelper.DefaultTurtleExtension);
221 var format = MimeTypesHelper.GetDefinitionsByFileExtension(formatStr).Concat(MimeTypesHelper.GetDefinitions(formatStr)).FirstOrDefault(f => sparql ? f.CanWriteSparqlResults : (f.CanWriteRdf || f.CanWriteRdfDatasets));
222 if(format == null)
223 {
224 throw new ApplicationException($"Format '{options.Format}' is not recognized or could not be used for writing!");
225 }
226 OutputLog.WriteLine($"Using {format.SyntaxName + (options.CompressedOutput ? " (compressed)" : "")} for output.");
227
228 Func<Stream> formatOutputFactory = () => outputFactory(GetOutputMimeType(format));
229
230 if(sparql)
231 {
232 if(options.DirectOutput)
233 {
234 await WriteSparqlToOutput(entities, queries, graphHandlers, formatOutputFactory, format, options);
235 }else{
236 await SaveSparqlToOutput(entities, queries, graphHandlers, formatOutputFactory, format, options);
237 }
238 }else{
239 if(options.DirectOutput)
240 {
241 await WriteRdfToOutput(entities, queries, graphHandlers, formatOutputFactory, format, options);

243 await SaveRdfToOutput(entities, queries, graphHandlers, formatOutputFactory, format, options);
244 }
245 }
246
247 OutputLog.WriteLine("Done!");

397 private async Task<IReadOnlyCollection<SparqlQuery>> GetQueries(InspectorOptions options)
398 {
399 SparqlQueryParser? queryParser = null;
400 var results = new List<SparqlQuery>();
401 foreach(var file in options.Queries)
402 {
403 if(queryParser == null)
404 {
405 queryParser = new SparqlQueryParser(SparqlQuerySyntax.Extended);
406 queryParser.DefaultBaseUri = new Uri(options.Root, UriKind.Absolute);
407 queryParser.Warning += OutputLog.WriteLine;
408 }
409 SparqlQuery query;
410 try{
411 try{
412 using var stream = file.Open();
413 using var reader = new StreamReader(stream, Encoding.UTF8, true, 4096, true);
414 query = queryParser.Parse(reader);
415 }catch(NotSupportedException)
416 {
417 // If synchronous reading is not available
418 using var buffer = new MemoryStream();
419 using(var stream = file.Open())
420 {
421 await stream.CopyToAsync(buffer);
422 }
423 buffer.Position = 0;
424 using var reader = new StreamReader(buffer, Encoding.UTF8, true, 4096, true);
425 query = queryParser.Parse(reader);
426 }
427 }catch(Exception e)
428 {
429 throw new ApplicationException($"Error while parsing SPARQL query in {file.Name}: {e.Message}", e);
430 }
431 if(options.OutputIsSparqlResults)
432 {
433 switch(query.QueryType)
434 {
435 case SparqlQueryType.Ask:
436 case SparqlQueryType.Select:
437 case SparqlQueryType.SelectAll:
438 case SparqlQueryType.SelectAllDistinct:
439 case SparqlQueryType.SelectDistinct:
440 case SparqlQueryType.SelectReduced:
441 results.Add(query);
442 continue;
443 }
444 }else{
445 switch(query.QueryType)
446 {
447 case SparqlQueryType.Construct:
448 results.Add(query);
449 continue;
450 case SparqlQueryType.Select:
451 case SparqlQueryType.SelectAll:
452 case SparqlQueryType.SelectAllDistinct:
453 case SparqlQueryType.SelectDistinct:
454 case SparqlQueryType.SelectReduced:
455 if(!query.Variables.Any(v => FileNodeQueryTester.NodeVariableName.Equals(v.Name)))
456 {
457 throw new ApplicationException($"The SELECT query in {file.Name} does not use the ?{FileNodeQueryTester.NodeVariableName} variable, which is necessary to use to match files for extraction.");
458 }
459 results.Add(query);
460 continue;
461 }
462 }
463 throw new ApplicationException($"Query in {file.Name} has an unsupported type ({query.QueryType}).");
464 }
465 return results;
