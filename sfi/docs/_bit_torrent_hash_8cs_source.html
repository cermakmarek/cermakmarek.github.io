106 private async ValueTask<BDictionary> CreateDictionary(IFileNodeInfo fileNode, bool content)
107 {
108 var dict = new BDictionary();
109 byte[] buffer;
110 dict["piece length"] = new BNumber(BlockSize);
111 dict["name"] = new BString(content ? fileNode.Name : "");
112 switch(fileNode)
113 {
114 case IFileInfo file:
115 var info = await BitTorrentHashCache.GetCachedInfo(BlockSize, file);
116 dict["length"] = new BNumber(file.Length);
117 buffer = new byte[info.BlockHashes.Count * HashAlgorithm.GetHashSize(file.Length) + info.LastHash.Length];
118 for(int i = 0; i < info.BlockHashes.Count; i++)
119 {
120 info.BlockHashes[i].CopyTo(buffer, i * HashAlgorithm.GetHashSize(file.Length));
121 }
122 info.LastHash.CopyTo(buffer, buffer.Length - info.LastHash.Length);
123 break;
124 case IDirectoryInfo directory:
125 var files = new List<(BitTorrentHashCache.FileInfo cached, BList<BString> path)>();
126 long bufferLength = 0;
127 var root = new BList<BString>();
128 if(!content)
129 {
130 root.Add(fileNode.Name);
131 }
132 foreach(var (file, path) in FindFiles(directory, root))
133 {
134 var cachedInfo = await BitTorrentHashCache.GetCachedInfo(BlockSize, file);
135 files.Add((cachedInfo, path));
136 bufferLength += cachedInfo.BlockHashes.Count * HashAlgorithm.GetHashSize((file as IFileInfo)?.Length ?? 0) + cachedInfo.LastHashPadded.Length;
137 }
138 buffer = new byte[bufferLength];
139 int pos = 0;
140 var filesList = new BList<BDictionary>();
141 foreach(var (cachedInfo, path) in files)
142 {
143 var fileDict = new BDictionary();
144 fileDict["path"] = path;
145 fileDict["length"] = new BNumber(cachedInfo.Length);
146 filesList.Add(fileDict);
147
148 foreach(var hash in cachedInfo.BlockHashes)
149 {
150 hash.CopyTo(buffer, pos);
151 pos += hash.Length;
152 }
153 cachedInfo.LastHashPadded.CopyTo(buffer, pos);
154 pos += cachedInfo.LastHashPadded.Length;
155
156 if(cachedInfo.Padding > 0)
157 {
158 filesList.Add(new BDictionary
159 {
160 { "path", new BList<BString>{ ".pad", cachedInfo.Padding.ToString() } },
161 { "length", new BNumber(cachedInfo.Padding) }
162 });
163 }
164 }
165 dict["files"] = filesList;
166 break;
167 default:
168 throw new NotSupportedException();
169 }
170 dict["pieces"] = new BString(buffer);
171 return dict;
