207 public async ValueTask<AnalysisResult> Analyze<TEntity>(TEntity entity, AnalysisContext context) where TEntity : class
208 {
209 List<ContainerAnalysisInfo>? followedAnalyzers = null;
210 List<IContainerAnalyzerProvider>? blocked = null;
211
212 // Start traversing the collection of active analyzers
213 var enumerator = activeAnalyzers.GetEnumerator();
214
215 // The implementation of AnalyzeInner
216 async ValueTask<AnalysisResult> GetResult(ContainerBehaviour behaviour)
217 {
218 if((behaviour & ContainerBehaviour.FollowChildren) != 0)
219 {
220 // The current analyzer should be followed into children
221 (followedAnalyzers ??= new List<ContainerAnalysisInfo>())
222 .Add(enumerator.Current);
223 }
224 if((behaviour & ContainerBehaviour.BlockOther) != 0)
225 {
226 // The current analyzer's provider is blocked for root matches
227 (blocked ??= new List<IContainerAnalyzerProvider>())
228 .Add(enumerator.Current.Provider);
229 }
230 // The parent is null only for the root of the hierarchy
231 var parent = Value != null ? this : null;
232 if(enumerator.MoveNext())
233 {
234 // There is a next analyzer; use it
235 return await enumerator.Current.Analyzer.Analyze(parent, entity, context, GetResult, analyzers);
236 }
237 // There are no more analyzers
238 if(followedAnalyzers == null && blocked == null && Value != null)
239 {
240 // No special handling required, just use the stored IEntityAnalyzers
241 return await analyzers.Analyze(entity, context);
242 }
243 var innerAnalyzer = analyzers;
244 if(parent != null)
245 {
246 // Analyzing a new node
247 innerAnalyzer = new ContainerNode<TEntity, IContainerNode<TValue, TParent>>(parent, entity, followedAnalyzers, baseProvider, innerAnalyzer);
248 // Try matching the node as a root for other analyzers
249 var wrapper = baseProvider.MatchRoot(entity, context, innerAnalyzer, blocked);
250 if(wrapper != null)
251 {
252 return await wrapper.Analyze(entity, context);
253 }
254 }else{
255 // Nested nodes will use this analyzer
256 innerAnalyzer = new ContainerNode<TEntity, IContainerNode>(parent, entity, followedAnalyzers, baseProvider, innerAnalyzer);
257 }
258 return await baseProvider.Analyze(entity, context, innerAnalyzer);
259 }
260
261 return await GetResult(0);
262 }
