29 public static string SizeSuffix(long value, int decimalPlaces)
30 {
31 if(value < 0) return "-" + SizeSuffix(-value, decimalPlaces);
32 if(value == 0) return String.Format(CultureInfo.InvariantCulture, $"{{0:0.{new string('#', decimalPlaces)}}} B", 0);
33
34 int n = (int)Math.Log(value, 1024);
35 decimal adjustedSize = (decimal)value / (1L << (n * 10));
36 if(Math.Round(adjustedSize, decimalPlaces) >= 1000)
37 {
38 n += 1;
39 adjustedSize /= 1024;
40 }
41 return String.Format(CultureInfo.InvariantCulture, $"{{0:0.{new string('#', decimalPlaces)}}} {{1}}", adjustedSize, units[n]);
42 }

65 static IEnumerable<string> GetUriMediaTypeComponents(Uri uri)
66 {
67 switch(uri.HostNameType)
68 {
69 case UriHostNameType.Dns:
70 // A domain is split into its individual domain names, starting from the TLD
71 if(!String.IsNullOrEmpty(uri.IdnHost))
72 {
73 foreach(var name in uri.IdnHost.Split(hostSplitChars).Reverse())
74 {
75 yield return name;
76 }
77 }
78 break;
79 case UriHostNameType.IPv4:
80 // An IPv4 is split into its component octets
81 if(!String.IsNullOrEmpty(uri.Host))
82 {
83 foreach(var component in uri.Host.Split(hostSplitChars))
84 {
85 yield return component;
86 }
87 }
88 break;
89 case UriHostNameType.Unknown:
90 break;
91 default:
92 // Any other type of host is returned as is
93 if(!String.IsNullOrEmpty(uri.Host))
94 {
95 yield return uri.Host;
96 }
97 break;
98 }
99 // The scheme and port are next
100 yield return uri.Scheme;
101 if(!uri.IsDefaultPort)
102 {
103 yield return uri.Port.ToString();
104 }
105 // The path is split by : or / based on the presence of the host (it usually corresponds)
106 foreach(var segment in uri.AbsolutePath.Split(uri.HostNameType != UriHostNameType.Unknown ? slashSplitChars : colonSplitChars, StringSplitOptions.RemoveEmptyEntries))
107 {
108 yield return segment;
109 }
110 // Followed by the query and fragment
111 if(!String.IsNullOrEmpty(uri.Query))
112 {
113 yield return uri.Query;
114 }
115 if(!String.IsNullOrEmpty(uri.Fragment))
116 {
117 yield return uri.Fragment;
118 }
119 }

212 public static string GetTypeFriendlyName(Type type)
213 {
214 // Strip leading I for interfaces
215 string name = type.IsInterface ? interfaceLetter.Replace(type.Name, "") : type.Name;
216 var components = new List<string>();
217 var genArgs = type.GetGenericArguments();
218 int ownArgsStart = 0;
219 if(type.IsNested && !type.IsGenericParameter)
220 {
221 // Always include the name of the declaring type
222 var declaringType = type.DeclaringType;
223 if(declaringType.IsGenericTypeDefinition && type.IsGenericType)
224 {
225 // Nested type inherits generic parameters from declaring type
226 var genParams = declaringType.GetGenericArguments();
227 if(genArgs.Length > 0)
228 {
229 ownArgsStart = Math.Min(genArgs.Length, genParams.Length);
230 Array.Copy(genArgs, 0, genParams, 0, ownArgsStart);
231 declaringType = declaringType.MakeGenericType(genParams);
232 }
233 }
234 components.Add(GetTypeFriendlyName(declaringType));
235 }else{
236 if(!String.IsNullOrEmpty(type.Namespace))
237 {
238 // Get all similarly named visible types in the assembly
239 var similarTypes = GetExportedTypes(type.Assembly).Where(t => t.Name.Equals(type.Name, StringComparison.OrdinalIgnoreCase) && !t.Equals(type));
240 // Get the length of the namespace prefix shared by all these types
241 int prefix = similarTypes.Select(t => CommonPrefix(t.Namespace, type.Namespace)).DefaultIfEmpty(type.Namespace.Length).Max();
242 // Prepend the determining part of the namespace to the name
243 name = (prefix == type.Namespace.Length ? "" : type.Namespace.Substring(prefix)) + name;
244 }
245 // Produce components from the encoded name and its generic arguments
246 if(!friendNamespace.IsMatch(type.Namespace))
247 {
248 // If this is in an external assembly, distinguish it by the assembly name
249 components.Add(FormatMimeName(type.Assembly.GetName().Name));
250 }
251 }
252 // Strip the arity
253 int index = name.IndexOf('`');
254 if(index != -1) name = name.Substring(0, index);
255 // Add the base name and generic arguments
256 components.Add(FormatMimeName(name));
257 components.AddRange(genArgs.Skip(ownArgsStart).Select(GetTypeFriendlyName));
258 return String.Join(".", components);
259 }

353 public static string GetUserFriendlyName<T>(T entity)
354 {
355 if(entity == null) return "<null>";
356
357 if(entity is Type type)
358 {
359 if(!type.IsGenericType) return type.Name;
360 var sb = new StringBuilder();
361 var typeName = type.Name;
362 var index = typeName.IndexOf('`');
363 if(index != -1)
364 {
365 sb.Append(type.Name, 0, index);
366 }else{
367 sb.Append(type.Name);
368 }
369 sb.Append("<");
370 bool first = true;
371 foreach(var typeArg in type.GetGenericArguments())
372 {
373 if(first)
374 {
375 first = false;
376 }else{
377 sb.Append(", ");
378 }
379 sb.Append(GetUserFriendlyName(typeArg));
380 }
381 sb.Append(">");
382 return sb.ToString();
383 }
384

386 {
387 // trust IFormattable types to always provide a useful result
388 return formattable.ToString(null, CultureInfo.InvariantCulture);
389 }
390 string name;
391 type = entity.GetType();

393 if(TypeConverterHasDefinedStringConversion(converter))
394 {
395 // the converter is viable to convert to string
396 var convertedName = converter.ConvertToInvariantString(entity);
397 if(GetTypeFromName<T>(type, convertedName) is not Type existingTypeFromConverter)
398 {
399 // not the name of any type, so the converter or instance did actually provide its own string conversion
400 return convertedName;
401 }
402 name = entity.ToString();
403 if(name == convertedName)
404 {
405 // the converter just used the ToString on the entity, and we already know it corresponds to a type
406 return GetUserFriendlyName(existingTypeFromConverter);
407 }
408 }else{
409 name = entity.ToString();
410 }
411
412 // it is not viable to check whether ToString is overridden,
413 // since wrapper types may just redirect it to another instance,
414 // in which case the type would have to be found anyway
415 if(GetTypeFromName<T>(type, name) is Type existingType)
416 {
417 // it is the name of a type, so use that type instead
418 return GetUserFriendlyName(existingType);
419 }
420 return name;
421 }
