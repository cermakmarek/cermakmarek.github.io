128 public ResourceInfo(IModuleResource resource, int? ordinal)
129 {

135 if(TypeCode == Win32ResourceType.Bitmap || TypeCode == Win32ResourceType.Icon)
136 {
137 start = 14;
138 }
139 if(TypeCode == Win32ResourceType.Cursor)
140 {
141 start = 10;
142 }
143 var buffer = new byte[start + resource.Length];

222 public void MakeGroup(IReadOnlyDictionary<(object, object), ResourceInfo> cache)
223 {
224 var data = Data.AsSpan();
225 var header = data.MemoryCast<short>();
226 var type = header[1];
227 var count = header[2];
228 object typeCode;
229 switch(type)
230 {
231 case 1:
232 typeCode = Win32ResourceType.Icon;
233 break;
234 case 2:
235 typeCode = Win32ResourceType.Cursor;
236 break;
237 default:
238 return;
239 }
240
241 using var buffer = new MemoryStream();
242
243 int dataOffset = 6 + count * 16;
244 var resources = new List<ResourceInfo>();
245 var writer = new BinaryWriter(buffer);
246 buffer.Write(Data.Array, Data.Offset, 6);
247 for(int i = 0; i < count; i++)
248 {
249 var info = Data.Slice(6 + i * 14, 14);
