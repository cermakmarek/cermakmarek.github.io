330 public static IEnumerable<XmlReaderState> ReadFrom(XmlReader reader)
331 {
332 var namespaceFrames = new Stack<IReadOnlyDictionary<string, string>?>();
333 int depth = -1;
334 IReadOnlyDictionary<string, string>? namespaceMap = null;
335
336 while(reader.Read())
337 {
338 if(reader.Depth > depth)
339 {
340 namespaceFrames.Push(namespaceMap);
341 }else if(reader.Depth <= depth)
342 {
343 namespaceMap = namespaceFrames.Peek();
344 }
345 depth = reader.Depth;
346
347 var state = new XmlReaderState(reader, namespaceMap);
348 yield return state;
349
350 namespaceMap = state.NamespaceMap;
351 }
352
353 yield return new XmlReaderState(reader, null);
354 }
