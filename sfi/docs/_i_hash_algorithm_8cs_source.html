224 public virtual int EstimateUriSize(int hashSize)
225 {
226 var prefix = Prefix.Length;
227 switch(FormattingMethod)
228 {
229 case FormattingMethod.Hex:
230 return prefix + hashSize * 2;
231 case FormattingMethod.Base32:
232 return prefix + (hashSize + 4) / 5 * 8;
233 case FormattingMethod.Base58:
234 return prefix + (int)Math.Ceiling(hashSize * log58byte);
235 case FormattingMethod.Base64:
236 return prefix + (hashSize + 2) / 3 * 4;
237 case FormattingMethod.Decimal:
238 return prefix + (int)Math.Ceiling(hashSize * log10byte);
239 default:
240 throw new NotSupportedException();
241 }
242 }

253 public Uri this[ArraySegment<byte> data] {
254 get {
255 var sb = new StringBuilder(EstimateUriSize(data.Count));
256 sb.Append(Prefix);
257 if(data.Count > 0)
258 {
259 switch(FormattingMethod)
260 {
261 case FormattingMethod.Hex:
262 foreach(byte b in data)
263 {
264 sb.Append(b.ToString("X2"));
265 }
266 break;
267 case FormattingMethod.Base32:
268 DataTools.Base32(data, sb);
269 break;
270 case FormattingMethod.Base58:
271 DataTools.Base58(data, sb);
272 break;
273 case FormattingMethod.Base64:
274 DataTools.Base64Url(data, sb);
275 break;
276 case FormattingMethod.Decimal:
277 switch(data.Count)
278 {
279 case sizeof(byte):
280 sb.Append(data.Array[data.Offset]);
281 break;
282 case sizeof(ushort):
283 sb.Append(BitConverter.ToUInt16(data.Array, data.Offset));
284 break;
285 case sizeof(uint):
286 sb.Append(BitConverter.ToUInt32(data.Array, data.Offset));
287 break;
288 case sizeof(ulong):
289 sb.Append(BitConverter.ToUInt64(data.Array, data.Offset));
290 break;
291 default:
292 var dataCopy = new byte[data.Count + 1];
293 data.CopyTo(dataCopy, 0);
294 sb.Append(new BigInteger(dataCopy).ToString());
295 break;
296 }
297 break;
298 default:
299 throw new NotSupportedException();
300 }
301 }
302 return new Uri(sb.ToString());
303 }

330 public static async ValueTask<ILinkedNode?> AddHash(ILinkedNode node, IHashAlgorithm? algorithm, ArraySegment<byte> hash, ILinkedNodeFactory nodeFactory, OutputFileDelegate? output)
331 {
332 if(algorithm == null || hash.Count == 0) return null;
333
334 var hashNode = nodeFactory.Create(algorithm, hash);
335
336 hashNode.SetClass(Classes.Digest);
337
338 hashNode.Set(Properties.DigestAlgorithm, algorithm.Identifier);
339 hashNode.Set(Properties.DigestValue, hash.ToBase64String(), Datatypes.Base64Binary);
340
341 node.Set(Properties.Digest, hashNode);

343 if(algorithm is IEntityOutputProvider<byte[]> descProvider && output != null)
