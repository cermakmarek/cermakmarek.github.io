48 // is not root

51

73 node.Set(Properties.PrefLabel, DataTools.ReplaceControlCharacters(label, null));

78 node.Set(Properties.FileName, info.Name);
79 }
80
81 if(info.CreationTime is DateTime dt1)

83 node.Set(Properties.FileCreated, dt1);

87 node.Set(Properties.FileLastModified, dt2);

91 node.Set(Properties.FileLastAccessed, dt3);

126 public async ValueTask<AnalysisResult> Analyze(IFileInfo file, AnalysisContext context, IEntityAnalyzers analyzer)
127 {

129 if(node != null)
130 {
131 if(file.Length >= 0)
132 {
133 node.Set(Properties.FileSize, file.Length);
134 }
135

137 {

141 if(content != null)

143 node.Set(Properties.InterpretedAs, content);

145 }

147 foreach(var alg in HashAlgorithms)

154 await AnalyzeDirectory(node, directory, context, analyzer);

188 private async ValueTask<ILinkedNode?> AnalyzeContents(ILinkedNode? parent, IDirectoryInfo directory, AnalysisContext context, IEntityAnalyzers analyzer)
189 {
190 var folder = parent?[""] ?? context.NodeFactory.CreateUnique();
191
192 if(folder != null)
193 {
194 folder.SetClass(Classes.Folder);
195
196 parent?.Set(Properties.InterpretedAs, folder);
197
198 folder.Set(Properties.PrefLabel, DataTools.ReplaceControlCharacters(directory.ToString() + "/", null));
199
200 LinkDirectories(folder, directory.Path, true, context.NodeFactory);
201
202 foreach(var alg in HashAlgorithms)
203 {
204 await HashAlgorithm.AddHash(folder, alg, await alg.ComputeHash(directory, true), context.NodeFactory, OnOutputFile);
205 }
206
207 context = context.WithParent(parent);
208
209 foreach(var entry in directory.Entries)
210 {
211 var entryNode = CreateNode(entry, context);
212
213 await analyzer.Analyze(entry, context.WithNode(entryNode));
214 entryNode.Set(Properties.BelongsToContainer, folder);
215 }
216 }
217
218 return folder;

223 private void LinkDirectories(ILinkedNode initial, string? path, bool directory, ILinkedNodeFactory nodeFactory)
224 {
225 if(path == null) return;
226 var parts = path.Split('/');
227 if(parts.Length == 0) return;
228 for(int i = 0; i < parts.Length; i++)
229 {
230 var escapedParts = parts.Skip(i).Select(Uri.EscapeDataString);
231 if(directory)
232 {
233 escapedParts = escapedParts.Concat(directoryEnding);
234 }
235 var local = String.Join("/", escapedParts);
236 ILinkedNode file;
237 if(directory && local == "/")
238 {
239 file = nodeFactory.Create(RootDirectoryUri.Instance, default);
240 }else{
241 file = nodeFactory.Create(Vocabularies.File, local);
242 }
243 initial.Set(Properties.PathObject, file);
244 initial = file;
245 }
246
247 if(!directory)
248 {
249 string? ext;
250 try{
251 ext = Path.GetExtension(parts[parts.Length - 1]);
252 }catch(ArgumentException)
253 {
254 ext = null;
255 }
256 if(!String.IsNullOrEmpty(ext))
257 {
258 ext = ext!.Substring(1).ToLowerInvariant();
259 initial.Set(Properties.ExtensionObject, Vocabularies.Uris, Uri.EscapeDataString(ext));
260 }
261 }
