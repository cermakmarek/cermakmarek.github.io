273 private void SetProperties(object component, string componentName, IDictionary<string, string> properties)
274 {
275 foreach(var prop in GetConfigurableProperties(component))
276 {
277 var name = TextTools.FormatMimeName(prop.Name);
278 if(properties.TryGetValue(name, out var value))
279 {
280 // The property is assigned
281 properties.Remove(name);
282 var converter = prop.Converter;
283 object? convertedValue = null;
284 Exception? conversionException = null;
285 try{
286 convertedValue = converter.ConvertFromInvariantString(value);
287 }catch(Exception e)
288 {
289 // Conversion failed (for any reason)
290 conversionException = e;
291 }
292 if(convertedValue == null && !(String.IsNullOrEmpty(value) && conversionException == null))
293 {
294 throw new ApplicationException($"Cannot convert value '{value}' for property {componentName}:{name} to type {TextTools.GetIdentifierFromType(prop.PropertyType)}!", conversionException);
295 }
296 try{
297 prop.SetValue(component, convertedValue);
298 }catch(Exception e)
299 {
300 throw new ApplicationException($"Cannot assign value '{value}' to property {componentName}:{name}: {e.Message}", e);
301 }
302 }
303 }
304 foreach(var pair in properties)
305 {
306 LogWriter?.WriteLine($"Warning: Property {componentName}:{pair.Key} was not found!");
307 }

561 protected override void OnOptionArgumentFound(string option, string? argument)
562 {
563 switch(option)
564 {
565 case "r":
566 case "root":
567 if(rootSpecified)
568 {
569 throw OptionAlreadySpecified(option);
570 }
571 if(!Uri.TryCreate(argument, UriKind.Absolute, out _))
572 {
573 throw new ApplicationException("The argument to option '" + option + "' must be a well-formed absolute URI.");
574 }
575 options.Root = argument!;
576 rootSpecified = true;
577 break;
578 case "i":
579 case "include":
580 componentMatchers.Add(new Matcher(true, argument!));
581 break;
582 case "x":
583 case "exclude":
584 componentMatchers.Add(new Matcher(false, argument!));
585 break;
586 case "h":
587 case "hash":
588 var match = TextTools.ConvertWildcardToRegex(argument!);
589 if(mainHash == null)
590 {
591 mainHash = match;
592 mainHashName = argument;
593 componentMatchers.Add(new Matcher(true, "data-hash:" + argument!, true));
594 }else{
595 throw OptionAlreadySpecified(option);
596 }
597 break;
598 case "s":
599 case "sparql-query":
600 queries.Add(argument!);
601 break;
602 case "f":
603 case "format":
604 if(options.Format != null)
605 {
606 throw OptionAlreadySpecified(option);
607 }
608 options.Format = argument!;
609 break;
610 default:
611 if(componentPropertyRegex.Match(option) is { Success: true } propMatch)
612 {
613 var componentId = propMatch.Groups[1].Value.ToLowerInvariant();
614 if(!componentProperties.TryGetValue(componentId, out var dict))
615 {
616 componentProperties[componentId] = dict = new(StringComparer.OrdinalIgnoreCase);
617 }
618 dict[propMatch.Groups[2].Value.ToLowerInvariant()] = argument!;
619 }
620 break;
