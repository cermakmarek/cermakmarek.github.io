310 public void Parse(string[] args)
311 {
312 bool operands = false;
313 for(int i = 0; i < args.Length; i++)
314 {
315 string arg = ProcessArg(args[i]);
316 if(operands)
317 {
318 OnOperandFound(arg);
319 }else if(arg == "--")
320 {
321 operands = true;
322 }else if(arg.StartsWith("--"))
323 {
324 int delim = arg.IndexOf('=');
325 if(delim != -1)
326 {
327 string name = arg.Substring(2, delim-2);
328 if(OnOptionFound(name) == OptionArgument.None)
329 {
330 throw ArgumentNotExpected(name);
331 }
332 string argument = arg.Substring(delim+1);
333 OnOptionArgumentFound(name, argument);
334 }else{
335 string name = arg.Substring(2);
336 switch(OnOptionFound(name))
337 {
338 case OptionArgument.Optional:
339 OnOptionArgumentFound(name, null);
340 break;
341 case OptionArgument.Required:
342 if(++i >= args.Length) throw ArgumentExpected(name);
343 OnOptionArgumentFound(name, ProcessArg(args[i]));
344 break;
345 }
346 }
347 }else if(arg.Length > 1 && arg[0] == '-' && IsOptionChar(arg[1]))
348 {
349 for(int j = 1; j < arg.Length; j++)
350 {
351 char opt = arg[j];
352 string argument = String.Join("", arg.Skip(j+1).TakeWhile(c => !IsOptionChar(c)));
353
354 switch(OnShortOptionFound(opt))
355 {
356 case OptionArgument.None:
357 if(argument.Length > 0) throw ArgumentNotExpected(opt);
358 break;
359 case OptionArgument.Optional:
360 if(argument.Length > 0)
361 {
362 OnShortOptionArgumentFound(opt, argument);
363 }else{
364 OnShortOptionArgumentFound(opt, null);
365 }
366 break;
367 case OptionArgument.Required:
368 if(argument.Length > 0)
369 {
370 OnShortOptionArgumentFound(opt, argument);
371 }else{
372 if(j+1 < arg.Length)
373 {
374 OnShortOptionArgumentFound(opt, arg.Substring(j+1));
375 j = arg.Length-1;
376 }else if(++i >= args.Length)
377 {
378 throw ArgumentExpected(opt);
379 }else{
380 OnShortOptionArgumentFound(opt, ProcessArg(args[i]));
381 }
382 }
383 break;
384 }
385 j += argument.Length;
386 }
387 }else{
388 if(OnOperandFound(arg) == OperandState.OnlyOperands)
389 {
390 operands = true;
391 }
392 }
